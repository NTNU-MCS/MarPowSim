\section{$p_{max}$}
\begin{align}
t_e=\frac{p+r(p^2+q^2)/v}{\omega} \\
\dot{\omega}=0 \\
0=-D \omega-t_e+u_{max} \\
\frac{p+r p^2}{v}=-D\omega^2+u_{max}\omega- r q^2/v\\
\text{Assume: }p\leq1\\
p_{max}=\frac{v}{1+r}\left(-D\omega^2+u_{max}\omega- r q^2/v\right)
\end{align}
